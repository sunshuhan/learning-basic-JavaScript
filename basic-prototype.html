概念,参考地址： https://github.com/creeperyang/blog/issues/9
原型链的尽头（root）是Object.prototype。所有对象均从Object.prototype继承属性
  • var a = { x: 1 }; a.__proto__ === Object.prototype //true
  • function a(xx) { 1 + 1; } a.prototype; //constructor, __proto__
  • function d(xx) { 3 + 1; } var cc = new d(); cc.__proto__; //constructor, __proto__
  • function e(xx) { 3 + 1; } var ff = new e(); ff.__proto__ === e.prototype //true
  • Object instanceof Function; //true
  • Function instanceof Object; //true
  • Function.prototype.__proto__ == Object.prototype; //true
  • var n = 1; n.__proto__; //Number...
  • var p = 1; p.__proto__.__proto__ === Object.prototype; //true
